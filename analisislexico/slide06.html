public class IdentifierAFD {
    enum State { Q0, Q1, ERROR }
    
    public Token recognize(String input, int pos) {
        State state = State.Q0;
        int start = pos;
        
        while (pos < input.length()) {
            char c = input.charAt(pos);
            state = transition(state, c);
            
            if (state == State.ERROR) break;
            pos++;
        }
        
        if (state == State.Q1) {
            String lexeme = input.substring(start, pos);
            return new Token(IDENTIFIER, lexeme);
        }
        return null;
    }
    
    private State transition(State current, char c) {
        switch (current) {
            case Q0:
                return Character.isLetter(c) ? 
                       State.Q1 : State.ERROR;
            case Q1:
                return Character.isLetterOrDigit(c) ? 
                       State.Q1 : State.ERROR;
            default:
                return State.ERROR;
        }
    }
}

public class StringAFD {
    enum State { Q0, Q1, Q2, ERROR }
    
    public Token recognize(String input, int pos) {
        State state = State.Q0;
        int start = pos;
        
        while (pos < input.length()) {
            char c = input.charAt(pos);
            state = transition(state, c);
            
            if (state == State.ERROR) break;
            if (state == State.Q2) {
                pos++; // Consume closing quote
                break;
            }
            pos++;
        }
        
        if (state == State.Q2) {
            String lexeme = input.substring(start, pos);
            return new Token(STRING, lexeme);
        }
        return null;
    }
    
    private State transition(State current, char c) {
        switch (current) {
            case Q0:
                return (c == '"') ? State.Q1 : State.ERROR;
            case Q1:
                return (c == '"') ? State.Q2 : State.Q1;
            case Q2:
                return State.ERROR; // Final state
            default:
                return State.ERROR;
        }
    }
}

    enum State { Q0, Q1, Q2, Q3, Q4, ERROR }
    
    public Token recognize(String input, int pos) {

        int lastAcceptingPos = pos;
        
        while (pos < input.length()) {
            char c = input.charAt(pos);
            state = transition(state, c);
            
            if (state == State.ERROR) break;
            
            // Maximal Munch: guardar √∫ltimo estado v√°lido
            if (isAccepting(state)) {

        if (lastAccepting != null) {
            String lexeme = input.substring(start, lastAcceptingPos);
            return createToken(lastAccepting, lexeme);
        }
        return null;
    }
    
    private State transition(State current, char c) {
        switch (current) {
            case Q0:
                if (c == '+') return State.Q1;
                if (c == '=') return State.Q3;
                return State.ERROR;
            case Q1: // PLUS
                return (c == '+') ? State.Q2 : State.ERROR;
            case Q3: // EQUAL  
                return (c == '=') ? State.Q4 : State.ERROR;
            default:
                return State.ERROR;
        }
    }
}
