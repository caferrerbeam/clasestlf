🎯 Implementación Java - Detector de Errores
public class SyntaxErrorHandler {
    private List<SyntaxError> errors = new ArrayList<>();
    private TokenStream tokens;
    
    // PASO 1: Detectar error durante parsing
    public StatementNode parseStatement() throws ParseException {
        Token current = tokens.current();
        
        try {
            switch (current.getType()) {
                case IF:
                    return parseIfStatement();
                case WHILE:
                    return parseWhileStatement();
                case IDENTIFIER:
                    return parseAssignment();
                default:
                    // ERROR DETECTADO AQUÍ
                    throw new ParseException(
                        createDetailedError(current, "statement")
                    );
            }
        } catch (ParseException e) {
            // PASO 2: Registrar error
            recordError(e);
            
            // PASO 3: Intentar recuperación
            return recoverFromError();
        }
    }
    
    // PASO 2: Crear reporte detallado
    private SyntaxError createDetailedError(Token token, String expected) {
        return new SyntaxError.Builder()
            .setLocation(token.getLine(), token.getColumn())
            .setMessage(String.format(
                "Se esperaba %s pero se encontró '%s'", 
                expected, token.getLexeme()))
            .setSourceLine(getSourceLine(token.getLine()))
            .setContext(getParsingContext())
            .setSuggestion(generateSuggestion(token, expected))
            .build();
    }
    
    // PASO 4: Estrategias de recuperación
    private StatementNode recoverFromError() {
        // Estrategia 1: Panic Mode
        syncToNextStatement();
        
        // Estrategia 2: Insertion Recovery
        if (canInsertSemicolon()) {
            insertVirtualSemicolon();
            return parseNextStatement();
        }
        
        // Estrategia 3: Skip and Continue
        return new ErrorStatementNode();
    }
}

Estrategia 1: Panic Mode Recovery
Descartar tokens hasta encontrar un punto de sincronización

📝 Ejemplo Problemático:

// Error: 'xyz' no es válido aquí
if (x > 5) xyz print("mayor");
while (i < 10) {
    i++;
}

⚡ Proceso de Recuperación:

1
Error Detectado
Token 'xyz' inesperado después de ')'
2
Descartar Tokens
Eliminar 'xyz' y 'print' hasta encontrar ';'
3
Reanudar
Continuar parsing desde 'while'

🔍 Visualización del Proceso:

✓ if (x > 5) xyz print ; ← Punto de sincronización
✓ while (i < 10) { i++; }
✅ Resultado:

Se reporta 1 error, pero el parser puede continuar y encontrar el resto del código correcto.

🎯 Implementación Panic Mode
private void panicModeRecover() {
    System.out.println("🚨 Error detectado - Iniciando Panic Mode");
    
    // Puntos de sincronización comunes
    Set<TokenType> syncTokens = Set.of(
        TokenType.SEMICOLON,    // ; - fin de statement
        TokenType.RBRACE,       // } - fin de bloque
        TokenType.IF,           // if - inicio de nuevo statement
        TokenType.WHILE,        // while - inicio de loop
        TokenType.FOR,          // for - inicio de loop
        TokenType.RETURN        // return - statement de retorno
    );
    
    int tokensDiscarded = 0;
    
    // Descartar tokens hasta encontrar sincronización
    while (!tokens.isAtEnd() && 
           !syncTokens.contains(tokens.current().getType())) {
        
        Token discarded = tokens.advance();
        tokensDiscarded++;
        System.out.println("🗑️ Descartando: " + discarded.getLexeme());
    }
    
    System.out.println("📍 Punto de sincronización encontrado: " + 
                      tokens.current().getLexeme());
    System.out.println("📊 Tokens descartados: " + tokensDiscarded);
    
    // Si encontramos ';', lo consumimos para estar en estado limpio
    if (tokens.check(TokenType.SEMICOLON)) {
        tokens.advance();
        System.out.println("✅ Semicolon consumido - Estado limpio");
    }
}

Estrategia 2: Insertion Recovery
Insertar tokens faltantes automáticamente

📝 Ejemplo Problemático:

// Error: Falta punto y coma
int x = 10
// Error: Falta paréntesis de cierre
if (y > 5 {
    print("mayor");
}

⚡ Proceso de Recuperación:

1
Analizar Contexto
¿Qué token debería estar aquí?
2
Insertar Token Virtual
Agregar ';' y ')' donde faltan
3
Continuar Normal
Parsear como si estuviera correcto

🔧 Código Corregido Automáticamente:

int x = 10; ← Insertado
if (y > 5) { ← Insertado
    print("mayor");
}
✅ Ventajas:
  • Parsing continúa sin interrupciones
  • Se detectan errores adicionales
  • Experiencia de usuario mejorada
🎯 Implementación Insertion Recovery
private boolean attemptInsertionRecovery() {
    Token current = tokens.current();
    System.out.println("🔧 Analizando contexto para inserción...");
    
    // Caso 1: Falta punto y coma después de declaración
    if (isEndOfStatement() && !tokens.check(TokenType.SEMICOLON)) {
        System.out.println("📍 Detectado: Falta ';' después de statement");
        
        // Crear token virtual
        Token virtualSemi = new Token(
            TokenType.SEMICOLON, ";",
            current.getLine(), current.getColumn()
        );
        
        // Insertar en el stream
        tokens.insertVirtual(virtualSemi);
        
        System.out.println("✅ Punto y coma insertado automáticamente");
        reportRecovery("Se insertó ';' automáticamente en línea " + 
                      current.getLine());
        return true;
    }
    
    // Caso 2: Falta paréntesis de cierre
    if (isInParenthesesContext() && 
        (tokens.check(TokenType.LBRACE) || tokens.check(TokenType.SEMICOLON))) {
        
        System.out.println("📍 Detectado: Falta ')' antes de " + 
                          current.getLexeme());
        
        tokens.insertVirtual(new Token(
            TokenType.RPAREN, ")",
            current.getLine(), current.getColumn()
        ));
        
        System.out.println("✅ Paréntesis de cierre insertado");
        reportRecovery("Se insertó ')' automáticamente");
        return true;
    }
    
    // Caso 3: Falta llave de apertura en bloque
    if (expectingBlockStart() && !tokens.check(TokenType.LBRACE)) {
        System.out.println("📍 Detectado: Falta '{' para iniciar bloque");
        
        tokens.insertVirtual(new Token(
            TokenType.LBRACE, "{",
            current.getLine(), current.getColumn()
        ));
        
        System.out.println("✅ Llave de apertura insertada");
        return true;
    }
    
    System.out.println("❌ No se pudo realizar inserción automática");
    return false;
}

Estrategia 3: Deletion Recovery
Eliminar tokens extra o duplicados

📝 Ejemplo Problemático:

// Error: Punto y coma extra
int x = 10;;
// Error: Operador duplicado
if (y >> 5) {
    print("mayor");
}
// Error: Paréntesis extra
while ((i < 10)) {
    i++;
}

⚡ Proceso de Recuperación:

1
Detectar Extra
Identificar tokens duplicados o innecesarios
2
Eliminar Token
Saltar el token problemático
3
Re-intentar
Continuar parsing sin el token extra

✂️ Código Limpiado Automáticamente:

int x = 10; ; ← Eliminado
if (y > > 5) { ← Eliminado
    print("mayor");
}
while ((i < 10)) { ← Eliminados
    i++;
}
✅ Casos Comunes:
  • Puntos y comas duplicados
  • Operadores repetidos (>>, ++, --)
  • Paréntesis extras innecesarios
  • Llaves redundantes
🎯 Implementación Deletion Recovery
private boolean attemptDeletionRecovery() {
    Token current = tokens.current();
    Token next = tokens.peek(1);
    
    System.out.println("✂️ Analizando tokens para eliminación...");
    
    // Caso 1: Punto y coma duplicado
    if (current.getType() == TokenType.SEMICOLON && 
        next.getType() == TokenType.SEMICOLON) {
        
        System.out.println("📍 Detectado: Punto y coma duplicado");
        tokens.advance(); // Saltar el extra
        
        System.out.println("✅ Punto y coma extra eliminado");
        reportRecovery("Se eliminó ';' duplicado en línea " + 
                      current.getLine());
        return true;
    }
    
    // Caso 2: Operadores duplicados
    if (isOperator(current) && 
        current.getLexeme().equals(next.getLexeme())) {
        
        System.out.println("📍 Detectado: Operador duplicado '" + 
                          current.getLexeme() + "'");
        tokens.advance(); // Saltar el duplicado
        
        System.out.println("✅ Operador duplicado eliminado");
        reportRecovery("Se eliminó operador duplicado '" + 
                      current.getLexeme() + "'");
        return true;
    }
    
    // Caso 3: Paréntesis extra innecesario
    if (current.getType() == TokenType.LPAREN) {
        // Verificar si hay paréntesis anidados innecesarios
        if (next.getType() == TokenType.LPAREN && 
            !isValidNestedParentheses()) {
            
            System.out.println("📍 Detectado: Paréntesis extra innecesario");
            tokens.advance(); // Saltar el extra
            
            System.out.println("✅ Paréntesis extra eliminado");
            return true;
        }
    }
    
    // Caso 4: Punto y coma antes de llave de cierre
    if (current.getType() == TokenType.SEMICOLON && 
        next.getType() == TokenType.RBRACE) {
        
        System.out.println("📍 Detectado: ';' innecesario antes de '}'");
        tokens.advance(); // Saltar el semicolon
        
        System.out.println("✅ Semicolon innecesario eliminado");
        return true;
    }
    
    System.out.println("❌ No se detectaron tokens para eliminar");
    return false;
}

🎯 Parser con Manejo Robusto de Errores
public class RobustParser {
    private TokenStream tokens;
    private List<SyntaxError> errors;
    private RecoveryStrategy currentStrategy;
    
    public ParseResult parse() {
        errors = new ArrayList<>();
        
        try {
            ProgramNode program = parseProgram();
            return new ParseResult(program, errors);
        } catch (FatalParseException e) {
            // Error irrecuperable
            errors.add(e.getError());
            return new ParseResult(null, errors);
        }
    }
    
    private StatementNode parseStatement() {
        int errorsBefore = errors.size();
        
        try {
            return parseStatementInternal();
        } catch (ParseException e) {
            // Registrar error
            SyntaxError error = createDetailedError(e);
            errors.add(error);
            
            // Intentar recuperación múltiple
            return attemptMultipleRecoveries(e);
        }
    }
    
    private StatementNode attemptMultipleRecoveries(ParseException e) {
        // Estrategia 1: Insertion Recovery
        if (attemptInsertionRecovery()) {
            try {
                return parseStatementInternal();
            } catch (ParseException nested) {
                // Falló, intentar siguiente estrategia
            }
        }
        
        // Estrategia 2: Deletion Recovery
        if (attemptDeletionRecovery()) {
            try {
                return parseStatementInternal();
            } catch (ParseException nested) {
                // Falló, intentar siguiente estrategia
            }
        }
        
        // Estrategia 3: Panic Mode (último recurso)
        panicModeRecover();
        return new ErrorStatementNode(e.getMessage());
    }
    
    private SyntaxError createDetailedError(ParseException e) {
        Token errorToken = e.getToken();
        
        return new SyntaxError.Builder()
            .setLocation(errorToken.getLine(), errorToken.getColumn())
            .setMessage(generateUserFriendlyMessage(e))
            .setSourceLine(getSourceLine(errorToken.getLine()))
            .setErrorSpan(calculateErrorSpan(errorToken))
            .setContext(getCurrentParsingContext())
            .setSuggestion(generateSmartSuggestion(e))
            .setSeverity(calculateSeverity(e))
            .build();
    }
    
    private String generateSmartSuggestion(ParseException e) {
        TokenType expected = e.getExpectedType();
        Token found = e.getToken();
        
        switch (expected) {
            case SEMICOLON:
                return "Agregar ';' al final de la declaración";
            case RPAREN:
                return "Cerrar paréntesis con ')'";
            case RBRACE:
                return "Cerrar bloque con '}'";
            case IDENTIFIER:
                if (isKeyword(found.getLexeme())) {
                    return String.format("'%s' es palabra reservada, usar otro nombre", 
                                       found.getLexeme());
                }
                return "Se esperaba un identificador (nombre de variable)";
            default:
                return String.format("Reemplazar '%s' con '%s'", 
                                   found.getLexeme(), expected.getDisplayName());
        }
    }
    
    // Métricas de calidad del manejo de errores
    public ErrorStatistics getErrorStatistics() {
        return new ErrorStatistics.Builder()
            .setTotalErrors(errors.size())
            .setRecoveredErrors(countRecoveredErrors())
            .setFalsePositives(countFalsePositives())
            .setMostCommonErrors(findMostCommonErrors())
            .build();
    }
}
