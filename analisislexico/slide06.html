🤖 AFD - Autómata Finito Determinista

💡 ¿Qué es un AFD?
Un Autómata Finito Determinista es una máquina de estados que reconoce patrones de manera determinística (sin ambigüedad) y eficiente (tiempo lineal O(n)).

🎮 AFDs Interactivos

🔤 AFD para Identificadores

Patrón: [a-zA-Z][a-zA-Z0-9]* • Estados: q₀ (inicial), q₁ (aceptación) • Acepta: variable, count, x1, myVar

🔢 AFD para Números

Patrón: [0-9]+ • Estados: q₀ (inicial), q₁ (aceptación) • Acepta: 123, 0, 9999

📝 AFD para Strings

Patrón: "[^"]*" • Estados: q₀ (inicial), q₁ (contenido), q₂ (aceptación) • Acepta: "hello", "123", ""

⚡ AFD para Operadores

Patrón: +, ++, =, == • Maximal Munch: "++" se prefiere sobre "+" • "==" se prefiere sobre "="

📝 Implementación en Java
public class IdentifierAFD {
    enum State { Q0, Q1, ERROR }
    
    public Token recognize(String input, int pos) {
        State state = State.Q0;
        int start = pos;
        
        while (pos < input.length()) {
            char c = input.charAt(pos);
            
            switch (state) {
                case Q0:  // Estado inicial
                    if (Character.isLetter(c) || c == '_') {
                        state = State.Q1;
                        pos++;
                    } else {
                        return null;  // No es identificador
                    }
                    break;
                    
                case Q1:  // Estado de aceptación
                    if (Character.isLetterOrDigit(c) || c == '_') {
                        pos++;  // Continúa en Q1
                    } else {
                        // Fin del identificador
                        String lexeme = input.substring(start, pos);
                        return new Token(TokenType.IDENTIFIER, lexeme);
                    }
                    break;
            }
        }
        
        // Fin del input
        if (state == State.Q1) {
            String lexeme = input.substring(start, pos);
            return new Token(TokenType.IDENTIFIER, lexeme);
        }
        return null;
    }
}

public class NumberAFD {
    enum State { Q0, Q1, Q2, Q3, ERROR }
    
    public Token recognize(String input, int pos) {
        State state = State.Q0;
        int start = pos;
        TokenType type = TokenType.INTEGER;
        
        while (pos < input.length()) {
            char c = input.charAt(pos);
            
            switch (state) {
                case Q0:  // Estado inicial
                    if (Character.isDigit(c)) {
                        state = State.Q1;
                        pos++;
                    } else {
                        return null;  // No es número
                    }
                    break;
                    
                case Q1:  // Entero (estado de aceptación)
                    if (Character.isDigit(c)) {
                        pos++;  // Continúa en Q1
                    } else if (c == '.') {
                        state = State.Q2;
                        pos++;
                        type = TokenType.FLOAT;
                    } else {
                        // Fin del entero
                        String lexeme = input.substring(start, pos);
                        return new Token(TokenType.INTEGER, lexeme);
                    }
                    break;
                    
                case Q2:  // Después del punto
                    if (Character.isDigit(c)) {
                        state = State.Q3;
                        pos++;
                    } else {
                        return null;  // Error: punto sin dígitos
                    }
                    break;
                    
                case Q3:  // Decimal (estado de aceptación)
                    if (Character.isDigit(c)) {
                        pos++;  // Continúa en Q3
                    } else {
                        // Fin del decimal
                        String lexeme = input.substring(start, pos);
                        return new Token(TokenType.FLOAT, lexeme);
                    }
                    break;
            }
        }
        
        // Fin del input - verificar estado de aceptación
        if (state == State.Q1) {
            String lexeme = input.substring(start, pos);
            return new Token(TokenType.INTEGER, lexeme);
        } else if (state == State.Q3) {
            String lexeme = input.substring(start, pos);
            return new Token(TokenType.FLOAT, lexeme);
        }
        return null;
    }
}
