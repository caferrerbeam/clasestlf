public class RecursiveDescentParser {
    private TokenStream tokenStream;
    private List<SyntaxError> errors;
    
    public RecursiveDescentParser(List<Token> tokens) {
        this.tokenStream = new TokenStream(tokens);
        this.errors = new ArrayList<>();
    }
    
    /**
     * Punto de entrada principal - parsea todo el programa
     * Program ‚Üí Statement*
     */
    public ProgramNode parseProgram() {
        List<StatementNode> statements = new ArrayList<>();
        
        while (!tokenStream.isAtEnd()) {
            try {
                StatementNode stmt = parseStatement();
                statements.add(stmt);
            } catch (ParseException e) {
                errors.add(e.getError());
                // Recuperaci√≥n de errores - sincronizar al siguiente statement
                recoverToNextStatement();
            }
        }
        
        return new ProgramNode(statements);
    }
    
    /**
     * Statement ‚Üí IfStatement | WhileStatement | AssignmentStatement | Block
     */
    private StatementNode parseStatement() throws ParseException {
        Token current = tokenStream.current();
        
        switch (current.getType()) {
            case IF:
                return parseIfStatement();
            case WHILE:
                return parseWhileStatement();
            case LBRACE:
                return parseBlock();
            case IDENTIFIER:
                return parseAssignmentStatement();
            default:
                throw new ParseException(
                    "Token inesperado: " + current.getLexeme() + 
                    ". Se esperaba 'if', 'while', '{' o un identificador",
                    current
                );
        }
    }
    
    /**
     * IfStatement ‚Üí "if" "(" Expression ")" Statement ("else" Statement)?
     */
    private IfStatementNode parseIfStatement() throws ParseException {
        tokenStream.consume(TokenType.IF, "Se esperaba 'if'");
        tokenStream.consume(TokenType.LPAREN, "Se esperaba '(' despu√©s de 'if'");
        
        ExpressionNode condition = parseExpression();
        
        tokenStream.consume(TokenType.RPAREN, "Se esperaba ')' despu√©s de la condici√≥n");
        
        StatementNode thenStatement = parseStatement();
        StatementNode elseStatement = null;
        
        // Parte opcional del else
        if (tokenStream.match(TokenType.ELSE)) {
            elseStatement = parseStatement();
        }
        
        return new IfStatementNode(condition, thenStatement, elseStatement);
    }
    
    /**
     * WhileStatement ‚Üí "while" "(" Expression ")" Statement
     */
    private WhileStatementNode parseWhileStatement() throws ParseException {
        tokenStream.consume(TokenType.WHILE, "Se esperaba 'while'");
        tokenStream.consume(TokenType.LPAREN, "Se esperaba '(' despu√©s de 'while'");
        
        ExpressionNode condition = parseExpression();
        
        tokenStream.consume(TokenType.RPAREN, "Se esperaba ')' despu√©s de la condici√≥n");
        
        StatementNode body = parseStatement();
        
        return new WhileStatementNode(condition, body);
    }
    
    /**
     * AssignmentStatement ‚Üí ID "=" Expression ";"
     */
    private AssignmentNode parseAssignmentStatement() throws ParseException {
        Token identifier = tokenStream.consume(TokenType.IDENTIFIER, 
                                              "Se esperaba un identificador");
        
        tokenStream.consume(TokenType.ASSIGN, "Se esperaba '=' despu√©s del identificador");
        
        ExpressionNode value = parseExpression();
        
        tokenStream.consume(TokenType.SEMICOLON, 
                           "Se esperaba ';' al final de la asignaci√≥n");
        
        return new AssignmentNode(identifier.getLexeme(), value);
    }
    
    /**
     * Block ‚Üí "{" Statement* "}"
     */
    private BlockNode parseBlock() throws ParseException {
        tokenStream.consume(TokenType.LBRACE, "Se esperaba '{'");
        
        List<StatementNode> statements = new ArrayList<>();
        
        while (!tokenStream.check(TokenType.RBRACE) && !tokenStream.isAtEnd()) {
            statements.add(parseStatement());
        }
        
        tokenStream.consume(TokenType.RBRACE, "Se esperaba '}' para cerrar el bloque");
        
        return new BlockNode(statements);
    }
    
    /**
     * Expression ‚Üí Term Expression'
     */
    private ExpressionNode parseExpression() throws ParseException {
        ExpressionNode left = parseTerm();
        return parseExpressionPrime(left);
    }
    
    /**
     * Expression' ‚Üí ("+" | "-") Term Expression' | Œµ
     */
    private ExpressionNode parseExpressionPrime(ExpressionNode left) throws ParseException {
        TokenType currentType = tokenStream.current().getType();
        
        if (currentType == TokenType.PLUS || currentType == TokenType.MINUS) {
            Token operator = tokenStream.advance();
            ExpressionNode right = parseTerm();
            
            // Crear nodo binario y continuar recursivamente
            BinaryOperationNode binaryOp = new BinaryOperationNode(
                operator.getLexeme(), left, right
            );
            
            return parseExpressionPrime(binaryOp);
        }
        
        // Caso Œµ (epsilon) - no hay m√°s operadores
        return left;
    }
    
    /**
     * Term ‚Üí Factor Term'
     */
    private ExpressionNode parseTerm() throws ParseException {
        ExpressionNode left = parseFactor();
        return parseTermPrime(left);
    }
    
    /**
     * Term' ‚Üí ("*" | "/") Factor Term' | Œµ
     */
    private ExpressionNode parseTermPrime(ExpressionNode left) throws ParseException {
        TokenType currentType = tokenStream.current().getType();
        
        if (currentType == TokenType.MULTIPLY || currentType == TokenType.DIVIDE) {
            Token operator = tokenStream.advance();
            ExpressionNode right = parseFactor();
            
            BinaryOperationNode binaryOp = new BinaryOperationNode(
                operator.getLexeme(), left, right
            );
            
            return parseTermPrime(binaryOp);
        }
        
        return left; // Caso Œµ
    }
    
    /**
     * Factor ‚Üí NUMBER | ID | "(" Expression ")"
     */
    private ExpressionNode parseFactor() throws ParseException {
        Token current = tokenStream.current();
        
        switch (current.getType()) {
            case NUMBER:
                tokenStream.advance();
                return new LiteralNode(Double.parseDouble(current.getLexeme()));
                
            case IDENTIFIER:
                tokenStream.advance();
                return new VariableNode(current.getLexeme());
                
            case LPAREN:
                tokenStream.advance(); // Consume '('
                ExpressionNode expression = parseExpression();
                tokenStream.consume(TokenType.RPAREN, 
                                   "Se esperaba ')' despu√©s de la expresi√≥n");
                return expression;
                
            default:
                throw new ParseException(
                    "Token inesperado en factor: " + current.getLexeme() + 
                    ". Se esperaba un n√∫mero, identificador o '('",
                    current
                );
        }
    }
    
    /**
     * Recuperaci√≥n de errores - sincronizar al siguiente statement
     */
    private void recoverToNextStatement() {
        Set<TokenType> syncTokens = Set.of(
            TokenType.SEMICOLON, TokenType.RBRACE,
            TokenType.IF, TokenType.WHILE, TokenType.IDENTIFIER
        );
        
        while (!tokenStream.isAtEnd() && 
               !syncTokens.contains(tokenStream.current().getType())) {
            tokenStream.advance();
        }
        
        // Si encontramos ';', lo consumimos
        if (tokenStream.check(TokenType.SEMICOLON)) {
            tokenStream.advance();
        }
    }
    
    // M√©todos de acceso
    public List<SyntaxError> getErrors() { return errors; }
    public boolean hasErrors() { return !errors.isEmpty(); }
}
