public class Token {
    private TokenType type;      // Tipo del token
    private String lexeme;        // Texto original
    private Object value;         // Valor (si aplica)
    private int line;            // L√≠nea en el c√≥digo
    private int column;          // Columna en el c√≥digo
    
    public Token(TokenType type, String lexeme, 
                 int line, int column) {
        this.type = type;
        this.lexeme = lexeme;
        this.line = line;
        this.column = column;
        this.value = extractValue(type, lexeme);
    }
    
    private Object extractValue(TokenType type, 
                                String lexeme) {
        switch(type) {
            case INTEGER:
                return Integer.parseInt(lexeme);
            case FLOAT:
                return Float.parseFloat(lexeme);
            case STRING:
                return lexeme.substring(1, 
                       lexeme.length() - 1);
            default:
                return lexeme;
        }
    }
}
