import java.util.HashMap;
import java.util.Map;

public class Lexer {
    private Map<String, TokenType> keywords;
    
    public Lexer() {
        keywords = new HashMap<>();
        keywords.put("if", TokenType.KEYWORD_IF);
        keywords.put("while", TokenType.KEYWORD_WHILE);
        keywords.put("return", TokenType.KEYWORD_RETURN);
        keywords.put("class", TokenType.KEYWORD_CLASS);
        keywords.put("public", TokenType.KEYWORD_PUBLIC);
    }
    
    public Token matchIdentifier(String lexeme) {
        // Primero verificamos si es palabra reservada
        if (keywords.containsKey(lexeme)) {
            return new Token(keywords.get(lexeme), lexeme);
        } else {
            // Si no, es un identificador normal
            return new Token(TokenType.IDENTIFIER, lexeme);
        }
    }
}
