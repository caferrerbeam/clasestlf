    private int position;
    
    // üìä TABLA DE S√çMBOLOS - Palabras reservadas

        Map.of(

            "else",   TokenType.ELSE_KEYWORD,
            "while",  TokenType.WHILE_KEYWORD,
            "for",    TokenType.FOR_KEYWORD,
            "int",    TokenType.INT_TYPE,

        this.position = 0;

    // ‚≠ê nextToken() devuelve UN SOLO token
    public Token nextToken() throws LexicalException {

    // üìã Consultar tabla de s√≠mbolos
    private Token checkSymbolTable(Token token) {

    }
